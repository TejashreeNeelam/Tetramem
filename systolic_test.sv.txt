`include "uvm_macros.svh"
import uvm_pkg::*;
`include "common_types.sv"
`include "mat_tr.sv"

class systolic_test extends uvm_test;
    `uvm_component_utils(systolic_test)

    systolic_env env;

    function new(string name, uvm_component parent);
      super.new(name, parent);
    endfunction

    virtual function void build_phase(uvm_phase phase);
      super.build_phase(phase);
      env = systolic_env::type_id::create("env", this);
    endfunction

    virtual task run_phase(uvm_phase phase);
      phase.raise_objection(this);

      mat_seq seq1, seq2;
      seq1 = mat_seq::type_id::create("seq1");
      seq2 = mat_seq::type_id::create("seq2");

      fork
        seq1.start(env.sa.seqr);
        seq2.start(env.ssa.seqr);
      join

      #100ns;
      phase.drop_objection(this);
    endtask
  endclass
