`include "uvm_macros.svh"
import uvm_pkg::*;
`include "common_types.sv"
`include "mat_tr.sv"

class systolic_agent extends uvm_component;
    `uvm_component_utils(systolic_agent)

    uvm_sequencer #(mat_tr) seqr;
    systolic_driver         drv;
    systolic_monitor        mon;
    uvm_analysis_port #(mat_tr) ap;

    function new(string name, uvm_component parent);
      super.new(name, parent);
      ap = new("ap", this);
    endfunction

    virtual function void build_phase(uvm_phase phase);
      super.build_phase(phase);
      seqr = uvm_sequencer#(mat_tr)::type_id::create("seqr", this);
      drv  = systolic_driver::type_id::create("drv", this);
      mon  = systolic_monitor::type_id::create("mon", this);
    endfunction

    virtual function void connect_phase(uvm_phase phase);
      super.connect_phase(phase);
      drv.seq_item_port.connect(seqr.seq_item_export);
      mon.ap.connect(ap);
    endfunction
  endclass
